    v-on:click="btnClick"
  >

<script src="vue.js"></script>
<script>
  new Vue({
    el: '#app',
    data: {
      active: true,
    },
    methods: {
      btnClick() {
        this.active = !this.active;
      }
    }
  });
</script>
